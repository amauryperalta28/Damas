158  public void draw(SpriteBatch spritebatch, Vector2 _currentMousePosition, DragAndDropController<Item> _dragAndDropController)
159  {
160  float opacity;

162 
163  // Dimensiones de las casillas 80px x 80px
164  const int _tileSize = 80;

166  //the square to draw (local variable to avoid creating a new variable per square)
167  Rectangle squareToDrawPosition = new Rectangle();
168 
169  //Objeto en el que se insertaran todas las posiciones de las fichas que pueden comer otras fichas
170  Red fichaPrueba = new Red(spritebatch, _whiteSquare, new Vector2(0, 0));
171 
172  /*Se determinan las fichas que pueden comer y se guardan */
173  foreach (var fichaAEvaluar in _dragAndDropController.Items)
174  {
175  // Se verifica de quien es el turno para colorear las casillas de las fichas que pueden comer
176  if (_dragAndDropController.fichaPuedeComer(fichaAEvaluar) && fichaAEvaluar.Color.Equals(Colores.Red) &&
177  manejadorDeTurnos.turnoJugadorRojo == true)
178  {
179  fichaPrueba.addJugadaParaComerFicha(fichaAEvaluar.Position);
180  fichaAEvaluar.removeJugadasParaComerFicha();
181  }
182  else if (_dragAndDropController.fichaPuedeComer(fichaAEvaluar) && fichaAEvaluar.Color.Equals(Colores.Black) &&
183  manejadorDeTurnos.turnoJugadorNegro == true)
184  {
185  fichaPrueba.addJugadaParaComerFicha(fichaAEvaluar.Position);
186  fichaAEvaluar.removeJugadasParaComerFicha();
187 
188  }
189 
190  }
191 
192  // Se recorre el arreglo de casillas
193  for (int y = 0; y < casillas.GetLength(0); y++)
194  {
195  for (int x = 0; x < casillas.GetLength(1); x++)
196  {
197  //figure out where to draw the square
198  squareToDrawPosition = new Rectangle((int)(x * _tileSize + posicion.X), (int)(y * _tileSize + posicion.Y), _tileSize, _tileSize);
199 
200  Vector2 squarePosition = new Vector2(x * _tileSize + posicion.X, y * _tileSize + posicion.Y);
201  //if we add the x and y value of the tile
202  //and it is even, we make it one third opaque

206  }
207  else

209  //otherwise it is one tenth opaque

227  spritebatch.Draw(_whiteSquare, squareToDrawPosition, colorToUse * opacity);
228
